 93    def cargarDataGcode(self, archivo: str, tipoArchivo: str):
 94        with codecs.open(archivo, "r", encoding="utf-8", errors="ignore") as fdata:
 95            info = {
 96                "material": -1.0,
 97                "tiempo": -1.0,
 98                "tipo": -1.0,
 99                "color": -1.0,
100            }
101
102            data = fdata.read()
103            lineas = data.split("\n")
104            if tipoArchivo == ".bgcode":
105                lineasDocumento = "\n".join(lineas[:30])
106            else:
107                lineasDocumento = data
108            # print(f"Primera lineas: {lineasDocumento}")
109            # print("-"*80)
110
111            buscarImpresora = re.search(r"printer_model=.*(MMU3)", lineasDocumento)
112
113            if buscarImpresora:
114                self.multiMaterial = True
115            else:
116                self.multiMaterial = False
117
118            print(f"MultiMaterial de impresora: {self.multiMaterial}")
119
120            if self.multiMaterial:
121                # Formato de la línea: printer_model=MMU3
122                # filament used [g]=81.79, 17.95, 41.29, 0.00, 0.00
123                buscarGramos = re.search(
124                    r"filament used \[g\]=([0-9.]+), ([0-9.]+), ([0-9.]+), ([0-9.]+), ([0-9.]+)",
125                    lineasDocumento,
126                )
127                if buscarGramos:
128                    info["material"] = (
129                        float(buscarGramos.group(1))
130                        + float(buscarGramos.group(2))
131                        + float(buscarGramos.group(3))
132                        + float(buscarGramos.group(4))
133                        + float(buscarGramos.group(5))
134                    )
135            else:
136                buscarGramos = re.search(
137                    r"filament used \[g\]\s?=\s?([0-9.]+)", lineasDocumento
138                )
139
140                if buscarGramos:
141                    info["material"] = float(buscarGramos.group(1))
142
143            print(f"Buscar Gramos : { info['material']}")
144
145            time_match = re.search(
146                r"estimated printing time \(normal mode\)\s?=\s?(([0-9]+)h )?([0-9]+)m ([0-9]+)s",
147                lineasDocumento,
148            )
149
150            if time_match:
151                hours = (
152                    int(time_match.group(2)) if time_match.group(2) else 0
153                )  # Si no hay horas, usar 0
154                minutes = int(time_match.group(3))
155                seconds = int(time_match.group(4))
156                info["tiempo"] = hours + minutes / 60 + seconds / 3600
157
158            return info

238    def calcularCostos(self):
239        """Calcular costos"""
240
241        data = ObtenerArchivo("data/costos.md")
242        if data is None:
243            print("Error fatal con data costos.md")

253            tiempoEnsamblaje = float(dataVentas.get("tiempo_ensamblaje"))
254        horaTrabajo = float(data.get("hora_trabajo"))
255        costoEmsanblado = (tiempoEnsamblaje / 60) * horaTrabajo
256
257        self.costoGramo = float(data.get("precio_filamento")) / 1000
258
259        self.costoFilamento = self.totalGramos * self.costoGramo
260        self.costoEficiencia = (
261            self.costoFilamento * float(data.get("eficiencia_material")) / 100
262        )
263
264        self.costoImpresora = int(data.get("costo_impresora"))
265        self.envioImpresora = int(data.get("envio_impresora"))
266        self.mantenimientoImpresora = int(data.get("mantenimiento_impresora"))
267        self.vidaUtil = int(data.get("vida_util"))
268        self.tiempoTrabajo = (int(data.get("tiempo_trabajo")) / 100) * 8760
269        self.consumoPotencia = int(data.get("consumo"))
270        self.costoElectricidad = float(data.get("costo_electricidad"))
271        self.errorFabricacion = float(data.get("error_fabricacion")) / 100
272
273        costoTotalImpresora = (
274            self.costoImpresora
275            + self.envioImpresora
276            + self.mantenimientoImpresora * self.vidaUtil
277        )
278
279        costoRecuperacionInversion = costoTotalImpresora / (
280            self.tiempoTrabajo * self.vidaUtil
281        )
282        costoHora = (self.consumoPotencia / 1000) * self.costoElectricidad
283        costoHoraImpresion = (costoRecuperacionInversion + costoHora) * (
284            1 + self.errorFabricacion
285        )
286
287        print(f"costo extras: {self.costoExtras}")
288        print(f"costo hora impresión: {costoHoraImpresion:.2f}")
289        print(f"Precio por gramo: {self.costoGramo}")
290
291        costoHoraImpresion = costoHoraImpresion * self.totalHoras
292        self.costoTotal = (
293            self.costoFilamento
294            + self.costoEficiencia
295            + costoHoraImpresion
296            + costoEmsanblado
297            + self.costoExtras
298        )
299
300        self.costoUnidad = self.costoTotal / self.cantidadModelo
301
302        for data in self.tablaCostos.rows:

309            elif data["nombre"] == "Eficiencia filamento":

 94    def cargarDataGcode(self, archivo: str, tipoArchivo: str):
 95        with codecs.open(archivo, "r", encoding="utf-8", errors="ignore") as fdata:
 96            info = {
 97                "material": -1.0,
 98                "tiempo": -1.0,
 99                "tipo": -1.0,
100                "color": -1.0,
101            }
102
103            data = fdata.read()
104            lineas = data.split("\n")
105            if tipoArchivo == ".bgcode":
106                lineasDocumento = "\n".join(lineas[:30])
107            else:
108                lineasDocumento = data
109            # print(f"Primera lineas: {lineasDocumento}")
110            # print("-"*80)
111
112            buscarImpresora = re.search(r"printer_model=.*(MMU3)", lineasDocumento)
113
114            if buscarImpresora:
115                self.multiMaterial = True
116            else:
117                self.multiMaterial = False
118
119            print(f"MultiMaterial de impresora: {self.multiMaterial}")
120
121            if self.multiMaterial:
122                # Formato de la línea: printer_model=MMU3
123                # filament used [g]=81.79, 17.95, 41.29, 0.00, 0.00
124                buscarGramos = re.search(
125                    r"filament used \[g\]=([0-9.]+), ([0-9.]+), ([0-9.]+), ([0-9.]+), ([0-9.]+)",
126                    lineasDocumento,
127                )
128                if buscarGramos:
129                    info["material"] = (
130                        float(buscarGramos.group(1))
131                        + float(buscarGramos.group(2))
132                        + float(buscarGramos.group(3))
133                        + float(buscarGramos.group(4))
134                        + float(buscarGramos.group(5))
135                    )
136            else:
137                buscarGramos = re.search(
138                    r"filament used \[g\]\s?=\s?([0-9.]+)", lineasDocumento
139                )
140
141                if buscarGramos:
142                    info["material"] = float(buscarGramos.group(1))
143
144            print(f"Buscar Gramos : { info['material']}")
145
146            time_match = re.search(
147                r"estimated printing time \(normal mode\)\s?=\s?(([0-9]+)h )?([0-9]+)m ([0-9]+)s",
148                lineasDocumento,
149            )
150
151            if time_match:
152                hours = (
153                    int(time_match.group(2)) if time_match.group(2) else 0
154                )  # Si no hay horas, usar 0
155                minutes = int(time_match.group(3))
156                seconds = int(time_match.group(4))
157                info["tiempo"] = hours + minutes / 60 + seconds / 3600
158
159            return info
